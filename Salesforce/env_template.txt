# Salesforce OAuth Client Credentials Configuration
# Copy this content to your .env file and update with your actual values

# Client ID from your Salesforce External Client App (Consumer Key)
SALESFORCE_CLIENT_ID=your_client_id_here

# Client Secret from your Salesforce External Client App (Consumer Secret)  
SALESFORCE_CLIENT_SECRET=your_consumer_secret_here

# Your full Salesforce URL (including https://)
# Example: https://mycompany.my.salesforce.com or https://mycompany--dev.sandbox.my.salesforce.com
SALESFORCE_DEV_URL=https://your_domain.my.salesforce.com
